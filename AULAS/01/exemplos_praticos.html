Exemplos Práticos de Comunicação Web

JavaScript: Requisição Síncrona
XMLHttpRequest em modo síncrono (bloqueante)
function buscarDadosSincrono() {
  const xhr = new XMLHttpRequest();
  // 'false' torna a requisição síncrona
  xhr.open('GET', 'https://api.exemplo.com/dados', false);
  // BLOQUEIA até receber resposta
  xhr.send();

  if (xhr.status === 200) {
    return JSON.parse(xhr.responseText);
  }
}
// Interface fica bloqueada durante execução
const dados = buscarDadosSincrono();
JavaScript: Requisição Assíncrona
Usando Promises e fetch API (não-bloqueante)
function buscarDadosAssincrono() {
  // fetch retorna uma Promise
  return fetch('https://api.exemplo.com/dados')
    .then(resposta => {
      if (!resposta.ok) throw new Error('Erro');
      return resposta.json();
    })
    .then(dados => {
      return dados;
    });
}

// A interface continua responsiva
buscarDadosAssincrono().then(dados => {
  // Processa quando disponíveis
});
Python: Requisição Síncrona
Biblioteca requests para requisições síncronas
import requests

def buscar_dados_sincrono():
  # Esta linha bloqueia até receber resposta
  resposta = requests.get('https://api.exemplo.com/dados')
  
  if resposta.status_code == 200:
    return resposta.json()

# Execução bloqueante
dados = buscar_dados_sincrono()
print(dados)
Python: Requisição Assíncrona
Usando asyncio e aiohttp para requisições assíncronas
import asyncio
import aiohttp

async def buscar_dados_assincrono():
  async with aiohttp.ClientSession() as session:
    async with session.get('https://api.exemplo.com/dados') as resposta:
      if resposta.status == 200:
        return await resposta.json()

# Execução não-bloqueante
async def main():
  dados = await buscar_dados_assincrono()

asyncio.run(main())
